
```dataview
TABLE WITHOUT ID
  月份 AS "月份",
  length(rows) AS "创建笔记数"
WHERE created AND created.year = 2025
GROUP BY dateformat(created, "yyyy-MM") AS 月份
SORT 月份
```


```dataviewjs
// 创建笔记日历
const pages = dv.pages().where(p => p.created && p.created.year === 2025);

if (pages.length > 0) {
  // 按月创建日历
  const months = [
    '一月', '二月', '三月', '四月', '五月', '六月',
    '七月', '八月', '九月', '十月', '十一月', '十二月'
  ];
  
  for (let month = 1; month <= 12; month++) {
    const monthStr = `2025-${month.toString().padStart(2, '0')}`;
    const monthPages = pages.where(p => 
      p.created.toFormat('yyyy-MM') === monthStr
    );
    
    if (monthPages.length > 0) {
      dv.header(3, `${months[month-1]} (${monthPages.length}篇)`);
      
      // 按日期分组
      const dayGroups = {};
      monthPages.forEach(p => {
        const day = p.created.day;
        if (!dayGroups[day]) dayGroups[day] = [];
        dayGroups[day].push(p.file.link);
      });
      
      // 创建日期列表
      const dayList = Object.entries(dayGroups)
        .sort(([a], [b]) => a - b)
        .map(([day, links]) => 
          `${day}日: ${links.slice(0, 3).join("、")}${links.length > 3 ? "..." : ""}`
        );
      
      dv.list(dayList);
    }
  }
}
```
